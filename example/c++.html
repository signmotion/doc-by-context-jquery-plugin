Example 2

#include <iostream>
#include <vector>

int main () {
  int foo[] = { 10, 20, 30, 40, 50 };
  vector<int> bar;

  for (auto it = std::begin(foo); it!=std::end(foo); ++it)
    bar.push_back( sqrt( *it ) );

  cout << "bar contains:";
  for (auto it = std::begin(bar); it!=std::end(bar); ++it)
    cout << ' ' << *it;
  cout << '\n';

  return 0;
}
